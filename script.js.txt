async function getStrongSignal() {
    const input = document.getElementById("periodInput").value;
    const signalValue = document.getElementById("signalValue");
    const dot = document.getElementById("dotColor");
    const btn = document.getElementById("resultBtn");

    if (!input) {
        alert("Please enter period number!");
        return;
    }

    // Loading State
    btn.innerHTML = "FETCHING API...";
    btn.disabled = true;
    signalValue.innerHTML = "CONNECTING...";

    try {
        // ডামি API কল (এখানে আপনি আপনার আসল API লিঙ্ক বসাবেন)
        const response = await fetch('https://jsonplaceholder.typicode.com/todos/1');
        
        // ১.২ সেকেন্ড পর রেজাল্ট দেখানো (রিয়েলিস্টিক ফিল দেওয়ার জন্য)
        setTimeout(() => {
            const randomNum = Math.floor(Math.random() * 10);
            const size = (randomNum >= 5) ? "BIG" : "SMALL";
            const isRed = (randomNum % 2 === 0);
            const colorCode = isRed ? "#ff4d4d" : "#2ecc71";

            signalValue.innerHTML = `${randomNum} (${size})`;
            signalValue.style.color = colorCode;
            dot.style.background = colorCode;
            dot.style.boxShadow = `0 0 15px ${colorCode}`;

            btn.innerHTML = "GET RESULT";
            btn.disabled = false;
        }, 1200);

    } catch (error) {
        alert("API Connection Error!");
        btn.innerHTML = "RETRY";
        btn.disabled = false;
    }
}
